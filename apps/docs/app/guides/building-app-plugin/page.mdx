**Prerequisites:**

- TypeScript fundamentals
- React basics
- Understanding of [Plugins](/concepts/plugins) concept
- Familiarity with [episode structure](/getting-started/first-episode)

**Time estimate:** 30-45 minutes

## Step 1: Set Up Plugin Structure

First, create the file structure for your plugin:

```
packages/apps-myapp/
├── src/
│   ├── index.ts              # Public exports
│   ├── plugin.ts             # Plugin definition
│   ├── types.ts              # State and event types
│   ├── reducer.ts            # Event handlers
│   ├── dsl/
│   │   └── track-builder.ts  # DSL authoring API
│   └── ui/
│       └── MyAppUI.tsx       # React components
├── package.json
└── tsconfig.json
```

```json
{
  "name": "@tokovo/apps-myapp",

  "main": "src/index.ts",
```

## Step 2: Define State and Event Types

Define what your app's state looks like and what events it handles.

**`src/types.ts`**

```typescript
// State stored in WorldState
export interface MyAppState {
  messages: Message[];
  typing: Record<string, boolean>; // username -> isTyping
}

export interface Message {
  id: string;
  from: "user" | "other";
  sender: string; // Name of sender
  content: string;
  timestamp: number;
}

// Events your plugin handles
export type MyAppEvent =
  | {
      type: "APP_SEND_MESSAGE";
      payload: {
        content: string;
      };
    }
  | {
      type: "APP_RECEIVE_MESSAGE";
      payload: {
        from: string;
        content: string;
      };
    }
  | {
      type: "APP_TYPING_START";
      payload: {
        user: string;
      };
    }
  | {
      type: "APP_TYPING_END";
      payload: {
        user: string;
      };
    };
```

**Key Concepts:**

- **State is minimal** - Only store what's essential; derive everything else
- **Events are discriminated unions** - TypeScript helps catch errors
- **Messages track both directions** - `from` field distinguishes sender vs receiver

## Step 3: Implement Event Handlers with Immer

Event handlers update your app's state immutably using Immer.

**`src/reducer.ts`**

```typescript
import { produce } from "immer";
import type { MyAppState, MyAppEvent, Message } from "./types";

// Helper to generate unique IDs
let messageIdCounter = 0;
const generateId = () => `msg_${messageIdCounter++}`;

export function createInitialState(): MyAppState {
  return {
    messages: [],
    typing: {},
  };
}

export function myAppReducer(
  state: MyAppState,
  event: MyAppEvent,
  eventTimestamp: number,
): MyAppState {
  return produce(state, (draft) => {
    switch (event.type) {
      case "APP_SEND_MESSAGE": {
        const newMessage: Message = {
          id: generateId(),
          from: "user",
          sender: "You",
          content: event.payload.content,
          timestamp: eventTimestamp,
        };
        draft.messages.push(newMessage);
        break;
      }

      case "APP_RECEIVE_MESSAGE": {
        const newMessage: Message = {
          id: generateId(),
          from: "other",
          sender: event.payload.from,
          content: event.payload.content,
          timestamp: eventTimestamp,
        };
        draft.messages.push(newMessage);
        break;
      }

      case "APP_TYPING_START": {
        draft.typing[event.payload.user] = true;
        break;
      }

      case "APP_TYPING_END": {
        delete draft.typing[event.payload.user];
        break;
      }
    }
  });
}
```

**Key Patterns:**

- **Immer's `produce()`** - Allows safe mutations on `draft`
- **Switch on event type** - Handle each event distinctly
- **Generate IDs deterministically** - Ensures reproducibility
- **Use event timestamp** - Store timing info from the event

## Step 5: Register Anchors for Camera Framing

Anchors allow the camera to focus on specific UI elements.

**Update `src/ui/MyAppUI.tsx`:**

```typescript
import React, { useLayoutEffect } from "react";
import type { MyAppState } from "../types";

interface MyAppUIProps {
  state: MyAppState;
  deviceId: string; // NEW: Device ID for anchor registration
}

export function MyAppUI({ state, deviceId }: MyAppUIProps) {
  // Register anchor for the last message
  useLayoutEffect(() => {
    const lastMsg = state.messages[state.messages.length - 1];
    if (lastMsg) {
      // Register anchor for camera framing
      // (Simplified - in real plugin, use registries.plugins.anchors)
      window.registerAnchor?.(`${deviceId}:lastMessage`, {
        x: 0.5, // Centered horizontally
        y: 0.6, // Slightly below center
        width: 0.7, // 70% of screen width
        height: 0.15, // 15% of screen height
      });
    }
  }, [state.messages, deviceId]);

  return (
    <div
      style={{
        display: "flex",
        flexDirection: "column",
        height: "100%",
        backgroundColor: "#f5f5f5",
        fontFamily: "system-ui, -apple-system, sans-serif",
      }}
    >
      {/* ...rest of UI... */}
    </div>
  );
}
```

**Anchor Concepts:**

- **Normalized coordinates** - x, y, width, height in range 0-1
- **Device-scoped anchors** - Prefix with deviceId for uniqueness
- **Layout-driven** - Register in `useLayoutEffect` before paint
- **Common anchors** - `lastMessage`, `typing`, `header`, `device`

In production, you'd use a scoped anchor registry:

```typescript
import { createTokovoRegistries } from "@tokovo/core";

const registries = createTokovoRegistries();

registries.plugins.anchors.register({
  appId: "app_myapp",
  framing: {},
  getAnchors: (_world, _layout, deviceId) => ({
    anchors: {
      [`${deviceId}:lastMessage`]: {
        x: 0.5,
        y: 0.6,
        width: 0.7,
        height: 0.15,
      },
    },
    deviceId,
    appId: "app_myapp",
  }),
});
```

## Step 6: Create DSL Builder for Authoring

The DSL builder provides a fluent API for authoring episodes.

**`src/dsl/track-builder.ts`**

```typescript
import type { TimelineEvent } from "@tokovo/core";
import type { MyAppEvent } from "../types";

export class MyAppTrackBuilder {
  private events: TimelineEvent[] = [];

  constructor(
    private fps: number,
    private deviceId: string,
    private getOrder: () => number,
  ) {}

  /**
   * Schedule an event at a specific time
   */
  at(time: string) {
    const seconds = this.parseTime(time);
    const frame = Math.round(seconds * this.fps);

    return {
      send: (content: string) => {
        this.addEvent(frame, {
          type: "APP_SEND_MESSAGE",
          payload: { content },
        });
        return this;
      },

      receive: (from: string, content: string) => {
        this.addEvent(frame, {
          type: "APP_RECEIVE_MESSAGE",
          payload: { from, content },
        });
        return this;
      },

      startTyping: (user: string) => {
        this.addEvent(frame, {
          type: "APP_TYPING_START",
          payload: { user },
        });
        return this;
      },

      stopTyping: (user: string) => {
        this.addEvent(frame, {
          type: "APP_TYPING_END",
          payload: { user },
        });
        return this;
      },
    };
  }

  /**
   * Convenience method for typing + auto-end
   */
  span(startTime: string, endTime: string) {
    return {
      typing: (user: string) => {
        this.at(startTime).startTyping(user);
        this.at(endTime).stopTyping(user);
        return this;
      },
    };
  }

  /**
   * Get all scheduled events
   */
  build(): TimelineEvent[] {
    return this.events.sort((a, b) => a.t - b.t);
  }

  // --- Private helpers ---

  private addEvent(frame: number, event: MyAppEvent) {
    this.events.push({
      t: frame,
      kind: "APP",
      deviceId: this.deviceId,
      appId: "app_myapp",
      order: this.getOrder(),
      ...event,
    });
  }

  private parseTime(time: string): number {
    if (time.endsWith("s")) {
      return parseFloat(time);
    }
    if (time.endsWith("ms")) {
      return parseFloat(time) / 1000;
    }
    // Assume frames if no unit
    return parseInt(time) / this.fps;
  }
}
```

**DSL Patterns:**

- **Fluent API** - Methods return `this` for chaining
- **Time parsing** - Accept `"1s"`, `"500ms"`, or raw frames
- **Automatic ordering** - Use `getOrder()` for deterministic event sequence
- **Type safety** - TypeScript ensures valid event payloads

## Step 7: Define the Plugin

Bundle everything together into a plugin definition.

**`src/plugin.ts`**

```typescript
import type { TokovoPlugin } from "@tokovo/core/plugin";
import { myAppReducer, createInitialState } from "./reducer";
import { MyAppUI } from "./ui/MyAppUI";
import type { MyAppState, MyAppEvent } from "./types";

export const MyAppPlugin: TokovoPlugin = {
  // Identity
  id: "app_myapp",
  version: "1.0.0",
  displayName: "My Messaging App",

  // Runtime: State management
  createInitialState,
  reducer: myAppReducer,

  // Runtime: UI
  views: {
    AppRoot: MyAppUI,
  },

  // Event routing (optional but recommended)
  eventKinds: [
    "APP_SEND_MESSAGE",
    "APP_RECEIVE_MESSAGE",
    "APP_TYPING_START",
    "APP_TYPING_END",
  ] as const,

  // Optional: Register audio rules, layouts, etc.
};
```

**Export the plugin:**

**`src/index.ts`**

```typescript
export { MyAppPlugin } from "./plugin";
```

Now you can use your plugin in episodes!

**Register the plugin:**

```typescript
// In your episode file or app entrypoint
import { createTokovoRegistries, PluginManagerClass } from "@tokovo/core";

const registries = createTokovoRegistries();

```

**Use it in an episode:**

```typescript
import { episode } from "@tokovo/dsl";
import { MyAppTrackBuilder } from "@tokovo/apps-myapp";

let order = 0;
const getOrder = () => order++;

episode("test-myapp", { fps: 30, duration: "20s" })
  .device("phone", "iphone16", {
    app: "app_myapp",
  })
  .track(
    "app_myapp",
    () => new MyAppTrackBuilder(30, "phone", getOrder),
    (app) => {
      // First message arrives
      app.at("1s").receive("Alice", "Hey! How's it going?");

      // Show typing indicator
      app.span("3s", "5s").typing("You");

      // Send a reply
      app.at("5s").send("Great! Just working on a new plugin.");

      // Alice starts typing
      app.span("7s", "10s").typing("Alice");

      // Alice responds
      app.at("10s").receive("Alice", "Cool! What does it do?");

      // Quick reply
      app.span("12s", "14s").typing("You");
      app.at("14s").send("It's a messaging app for Tokovo!");

      // Alice reacts
      app.at("16s").receive("Alice", "Nice! 🎉");
    },
  )
  .camera((cam) => {
    // Focus on messages as they arrive
    cam.at("1s").focus("lastMessage", { duration: "0.5s" });
    cam.at("10s").focus("lastMessage", { duration: "0.5s" });
    cam.at("16s").focus("lastMessage", { duration: "0.5s" });
  })
  .build();
```

## Common Patterns

### State Immutability

Always use Immer's `produce()` for state updates:

```typescript
// ✅ Good - Immer draft mutations
return produce(state, (draft) => {
  draft.messages.push(newMessage);
  draft.typing[user] = true;
});

// ❌ Bad - Direct mutations
state.messages.push(newMessage); // NEVER mutate directly!
```

### Anchor Registration Best Practices

- **Register in `useLayoutEffect`** - Before browser paint
- **Use device-scoped IDs** - Prefix with `deviceId`
- **Provide semantic names** - `lastMessage`, `typing`, `header`
- **Normalized coordinates** - 0-1 range for x, y, width, height

```typescript
// ✅ Good (inside your AnchorProvider.getAnchors)
return {
  anchors: {
    [`${deviceId}:lastMessage`]: {
      x: 0.5, // 50% from left
      y: 0.6, // 60% from top
      width: 0.7, // 70% of screen width
      height: 0.15, // 15% of screen height
    },
  },
  deviceId,
  appId: "app_myapp",
};

// ❌ Bad - pixel coordinates (non-responsive)
return {
  anchors: {
    lastMessage: {
      x: 540, // Breaks on different screen sizes
      y: 1200,
      width: 700,
      height: 150,
    },
  },
  deviceId,
  appId: "app_myapp",
};
```

### Event Type Naming Conventions

Follow Tokovo's naming patterns:

- **APP\_** prefix for app-level events
- **SCREAMING_SNAKE_CASE** for consistency
- **Verb-based names** - `SEND_MESSAGE`, `TYPING_START`

```typescript
// ✅ Good
"APP_SEND_MESSAGE";
"APP_TYPING_START";
"APP_RECEIVE_MESSAGE";

// ❌ Bad
"sendMessage"; // Wrong case
"MESSAGE"; // Missing verb
"send_msg"; // Inconsistent abbreviation
```

## Testing Your Plugin

### Unit Test the Reducer

```typescript
import { describe, it, expect } from "vitest";
import { myAppReducer, createInitialState } from "./reducer";

describe("MyApp Reducer", () => {
  it("handles APP_SEND_MESSAGE", () => {
    const state = createInitialState();
    const event = {
      type: "APP_SEND_MESSAGE" as const,
      payload: { content: "Hello" },
    };

    const nextState = myAppReducer(state, event, 1000);

    expect(nextState.messages).toHaveLength(1);
    expect(nextState.messages[0].content).toBe("Hello");
    expect(nextState.messages[0].from).toBe("user");
  });

  it("handles APP_TYPING_START and END", () => {
    let state = createInitialState();

    state = myAppReducer(
      state,
      { type: "APP_TYPING_START", payload: { user: "Alice" } },
      1000,
    );

    expect(state.typing["Alice"]).toBe(true);

    state = myAppReducer(
      state,
      { type: "APP_TYPING_END", payload: { user: "Alice" } },
      2000,
    );

    expect(state.typing["Alice"]).toBeUndefined();
  });
});
```

## Common Pitfalls to Avoid

### ❌ Storing Derived State

```typescript
// ❌ Bad - messageCount can be derived
interface MyAppState {
  messages: Message[];
  messageCount: number; // Redundant!
}

// ✅ Good - derive in component
const messageCount = state.messages.length;
```

### ❌ Direct State Mutations

```typescript
// ❌ Bad - mutating outside Immer
function reducer(state, event) {
  state.messages.push(newMessage); // BREAKS IMMUTABILITY!
  return state;
}

// ✅ Good - use Immer's produce
function reducer(state, event) {
  return produce(state, (draft) => {
    draft.messages.push(newMessage);
  });
}
```

### ❌ Hardcoding Device IDs

```typescript
// ❌ Bad - breaks multi-device support
return { anchors: { lastMessage: { /* ... */ } }, deviceId, appId: "app_myapp" };

// ✅ Good - device-scoped
return {
  anchors: { [`${deviceId}:lastMessage`]: { /* ... */ } },
  deviceId,
  appId: "app_myapp",
};
```

### ❌ Missing Event Ordering

```typescript
// ❌ Bad - events may execute out of order
this.events.push({ t: frame, ...event });

// ✅ Good - include order for determinism
this.events.push({ t: frame, order: this.getOrder(), ...event });
```
