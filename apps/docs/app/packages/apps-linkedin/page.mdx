# @tokovo/apps-linkedin

LinkedIn app simulation plugin for Tokovo. Includes reducer/runtime, UI views, v2 lowering, layouts, DSL track builder, and semantic anchors.

## Install

```bash
pnpm add @tokovo/apps-linkedin
```

## Register

```ts
import { createTokovoRegistries, PluginManagerClass } from "@tokovo/react";
import { registerLinkedInPlugin } from "@tokovo/apps-linkedin";

const registries = createTokovoRegistries();
const pluginManager = new PluginManagerClass(registries.plugins);

registerLinkedInPlugin(pluginManager);
```

## Use In Episodes

```ts
import { episode } from "@tokovo/dsl";
import { LinkedInTrackBuilder } from "@tokovo/apps-linkedin";

episode("linkedin-demo", { fps: 30, duration: "20s" })
  .device("phone", "iphone16", { app: "app_linkedin" })
  .track(
    "app_linkedin",
    (getOrder) => new LinkedInTrackBuilder(30, "phone", getOrder),
    (li) => {
      li.at("1s").createUser({ id: "me", name: "You", handle: "you", headline: "Builder" });
      li.at("1s").createUser({ id: "a", name: "Ava", handle: "ava", headline: "Product" });
      li.at("2s").post({ id: "p1", authorId: "a", text: "Ship with focus. Document with intent.", typed: true, charDelay: 2 });
      li.at("6s").react("p1", "me", "celebrate");
      li.at("8s").navigate("notifications");
    },
  )
  .build();
```
