# @tokovo/apps-snapchat

Snapchat app simulation plugin for Tokovo. Includes reducer/runtime, UI views, v2 lowering, layouts, anchors, and DSL track builders.

## Install

```bash
pnpm add @tokovo/apps-snapchat
```

## Register

```ts
import { createTokovoRegistries, PluginManagerClass } from "@tokovo/react";
import { registerSnapchatPlugin } from "@tokovo/apps-snapchat";

const registries = createTokovoRegistries();
const pluginManager = new PluginManagerClass(registries.plugins);

registerSnapchatPlugin(pluginManager);
```

## Use In Episodes

```ts
import { episode } from "@tokovo/dsl";
import { SnapchatTrackBuilder } from "@tokovo/apps-snapchat";

episode("snapchat-demo", { fps: 30, duration: "20s" })
  .device("phone", "iphone16", { app: "app_snapchat" })
  .track(
    "app_snapchat",
    (getOrder) => new SnapchatTrackBuilder(30, "phone", "friends", getOrder),
    (snap) => {
      snap.at("1s").openConversation("friends");
      snap.at("2s").receive("Ava", "Where are you?");
      snap.at("3s").typingStart("Ava");
      snap.at("4s").typingEnd("Ava");
      snap.at("5s").send("On my way.");
    },
  );
```
